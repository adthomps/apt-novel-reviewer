Schema:
type ReviewResult = {
  templateId: string;
  reviewType: string;
  status: "success" | "partial" | "error";
  model: string;

  summary: {
    overallAssessment: string;
    issueCount: number;
    highSeverityCount: number;
    mediumSeverityCount: number;
    lowSeverityCount: number;

  findings: Finding[];
  notes: Note[];

  generatedAt: string;

Finding Schema
type Finding = {
  id: string;
  type: string;
  severity: "low" | "medium" | "high";
  confidence: "low" | "medium" | "high";

  location: {
    chapterId: string;
    textAnchor: string;
  };

  issue: string;
  whyItMatters: string;
  evidence: string[];
  suggestedFix: string;

  status?: "new" | "still" | "resolved";
};
